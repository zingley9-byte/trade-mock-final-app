// Paste these rules into:
// Firebase Console → Firestore Database → Rules tab → Publish

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {

    // Admin config (announcements, coins, app settings)
    // Any authenticated user can read; only the admin email can write
    match /admin_config/{doc} {
      allow read: if request.auth != null;
      allow write: if request.auth != null
                   && request.auth.token.email == 'Zingley9@gmail.com';
    }

    // User documents — each user can only access their own data
    match /users/{uid} {
      allow read, write: if request.auth != null && request.auth.uid == uid;

      match /trades/{tradeId} {
        allow read, write: if request.auth != null && request.auth.uid == uid;
      }
    }
  }
}
